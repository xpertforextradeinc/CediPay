Integration Guide

Quick Setup

1. Include the CediPay widget script in your HTML:

<script src="https://your-domain.com/cedipay-checkout.js"></script>

2. Initialize with your merchant ID

3. Call CediPay.openCheckout() when ready to accept payment

Example Code

<!-- Include CediPay Widget -->
<script src="https://your-domain.com/cedipay-checkout.js"></script>

<script>
  // Initialize CediPay
  CediPay.init({
    merchantId: 'your-merchant-id-here',
    apiUrl: 'https://your-api-domain.com/api'
  });

  // Handle payment button click
  function handlePayment() {
    CediPay.openCheckout({
      amount: 100.00,
      customerName: 'John Doe',
      customerEmail: 'john@example.com',
      description: 'Purchase: Premium Product',
      onSuccess: function(data) {
        console.log('Payment successful!', data);
        // Update your backend, show confirmation, etc.
      },
      onError: function(error) {
        console.error('Payment failed:', error);
        // Show error message to user
      },
      onClose: function() {
        console.log('Payment modal closed');
      }
    });
  }
</script>

<button onclick="handlePayment()">Pay Now</button>
